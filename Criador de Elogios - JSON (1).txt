        <!-- Painel de Configuração -->

                <svg class="h-5 w-5 text-red-500 mr-2" viewBox="0 0 20 20" fill="currentColor">

    <script>
        /** * Mock do ambiente Google Apps Script para funcionamento no Preview
         */
        const apiKey = "";
        const google = {
            script: {
                run: {
                    withSuccessHandler: function(handler) {
                        this.success = handler;
                        return this;
                    },
                    withFailureHandler: function(handler) {
                        this.failure = handler;
                        return this;
                    },
                    generatePraise: async function(data) {
                        try {
                            const systemPrompt = "Você é um especialista em cultura organizacional e qualidade. Sua tarefa é transformar feedbacks brutos de clientes em posts entusiasmados para o Slack. Você deve extrair os pontos positivos e gerar um JSON com três campos: 'introducao' (uma frase curta de impacto), 'corpoElogio' (um resumo do feedback focando no excelente atendimento do analista) e 'textoRodape' (uma frase motivacional curta sobre o impacto positivo do atendimento). Responda estritamente em JSON e em Português do Brasil.";
                            const userQuery = `Empresa: ${data.empresa}\nAnalista: ${data.analista}\nFeedback do Cliente: ${data.elogio}`;

                            const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                                method: 'POST',
                                headers: { 'Content-Type': 'application/json' },
                                body: JSON.stringify({
                                    contents: [{ parts: [{ text: userQuery }] }],
                                    systemInstruction: { parts: [{ text: systemPrompt }] },
                                    generationConfig: { responseMimeType: "application/json" }
                                })
                            });

                            if (!response.ok) throw new Error("Erro na chamada da API.");
                            const result = await response.json();
                            const content = JSON.parse(result.candidates[0].content.parts[0].text);
                            this.success(content);
                        } catch (err) {
                            this.failure(err);
                        }
                    }
                }
            }
        };

        // Configurações visuais por empresa
        const configs = {
            Locaweb: { title: ":locaweb: *Elogio em Locaweb!* :locaweb:", emoji: "❤️", sig: ":omelhordenós:" },
            KingHost: { title: ":king: *Reconhecimento KingHost!* :king:", emoji: "💜", sig: ":omelhordenós:" },
            Octadesk: { title: "🐙 *Reconhecimento Octadesk!* 🐙", emoji: "💙", sig: ":omelhordenós:" }
        };

        const elements = {
            gerarBtn: document.getElementById('gerarBtn'),
            btnText: document.getElementById('btnText'),
            spinner: document.getElementById('spinner'),
            resultado: document.getElementById('resultado'),
            jsonOutput: document.getElementById('jsonOutput'),
            errorMsg: document.getElementById('errorMessage'),
            errorText: document.getElementById('errorText'),
            abrirSlack: document.getElementById('abrirSlackBtn'),
            copyBtn: document.getElementById('copyBtn'),
            copyText: document.getElementById('copyText')
        };

        let lastGeneratedJson = null;

        /**
         * Transforma links do Google Drive para visualização direta de imagem
         */
        function formatDriveUrl(url) {
            if (!url) return "https://via.placeholder.com/800x400?text=Sem+Imagem+de+Fundo";
            const regex = /drive\.google\.com\/(?:file\/d\/|uc\?id=)([a-zA-Z0-9_-]+)/;
            const match = url.match(regex);
            return match ? `https://drive.google.com/uc?export=view&id=${match[1]}` : url;
        }

        /**
         * Invoca o Block Kit Builder do Slack numa nova janela
         */
        function launchSlackBuilder(json) {
            const data = encodeURIComponent(JSON.stringify(json));
            const url = `https://app.slack.com/block-kit-builder/#${data}`;
            window.open(url, '_blank');
        }

        /**
         * Processamento após sucesso da IA no backend (Code.gs)
         */
        function handleSuccess(iaResult) {
            elements.spinner.classList.add('hidden');
            elements.gerarBtn.disabled = false;
            elements.btnText.textContent = "Gerar Post e Abrir no Slack";

            const empresa = document.getElementById('empresa').value;
            const analista = document.getElementById('analista').value.replace('@', '');
            const rawImg = document.getElementById('imageUrl').value;
            const config = configs[empresa];

            // Monta o corpo da mensagem com quebras de linha corretas
            const slackText = [
                config.title,
                iaResult.introducao,
                `Parabéns, @${analista}! ${config.emoji} ${iaResult.corpoElogio}`,
                `O @time-qualidade agradece e celebra a tua dedicação! ${config.sig}`
            ].join('\n\n');

            const finalJson = {
                blocks: [
                    {
                        type: "image",
                        image_url: formatDriveUrl(rawImg),
                        alt_text: "Post de Reconhecimento"
                    },
                    {
                        type: "section",
                        text: {
                            type: "mrkdwn",
                            text: slackText
                        }
                    },
                    { type: "divider" },
                    {
                        type: "context",
                        elements: [
                            {
                                type: "mrkdwn",
                                text: `⭐ ${iaResult.textoRodape}\n\n@here`
                            }
                        ]
                    }
                ]
            };

            lastGeneratedJson = finalJson;
            elements.jsonOutput.textContent = JSON.stringify(finalJson, null, 2);
            elements.resultado.classList.remove('hidden');
            
            // Tenta abrir o Slack automaticamente
            launchSlackBuilder(finalJson);
        }

        /**
         * Processamento em caso de falha na chamada ao Code.gs
         */
        function handleFailure(err) {
            elements.spinner.classList.add('hidden');
            elements.gerarBtn.disabled = false;
            elements.btnText.textContent = "Tentar Novamente";
            elements.errorText.textContent = err.message || "Ocorreu um erro ao processar o elogio.";
            elements.errorMsg.classList.remove('hidden');
        }

        // Clique no botão principal para invocar o Backend
        elements.gerarBtn.addEventListener('click', () => {
            const data = {
                empresa: document.getElementById('empresa').value,
                analista: document.getElementById('analista').value,
                elogio: document.getElementById('elogio').value
            };

            if (!data.analista || !data.elogio) {
                elements.errorText.textContent = "Por favor, preencha o utilizador do analista e o texto do elogio.";
                elements.errorMsg.classList.remove('hidden');
                return;
            }

            elements.spinner.classList.remove('hidden');
            elements.gerarBtn.disabled = true;
            elements.btnText.textContent = "Processando com IA...";
            elements.errorMsg.classList.add('hidden');

            // Comunicação direta com o ficheiro Code.gs (simulada aqui via Gemini API)
            google.script.run
                .withSuccessHandler(handleSuccess)
                .withFailureHandler(handleFailure)
                .generatePraise(data);
        });

        // Botão para reabrir no Slack manualmente
        elements.abrirSlack.addEventListener('click', () => {
            if (lastGeneratedJson) launchSlackBuilder(lastGeneratedJson);
        });

        // Funcionalidade de copiar JSON para a área de transferência
        elements.copyBtn.addEventListener('click', () => {
            const content = elements.jsonOutput.textContent;
            const el = document.createElement('textarea');
            el.value = content;
            document.body.appendChild(el);
            el.select();
            document.execCommand('copy');
            document.body.removeChild(el);
            
            elements.copyText.textContent = "JSON Copiado!";
            setTimeout(() => elements.copyText.textContent = "Copiar JSON", 2500);
        });
    </script>
